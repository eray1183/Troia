BEGINTRAN; 

LOOP AT EDUT2036TABLE1 WHERE EDUT2036TABLE1_UPDATED == 1 || EDUT2036TABLE1_DELETED == 1 || EDUT2036TABLE1_INSERTED == 1
BEGIN

	IF EDUT2036TABLE1_INSERTED == 1 && EDUT2036TABLE1_DELETED == 0 THEN
		INSERT INTO EDUT2036TABLE1;

		IF SYS_STATUS == 1 THEN
			ROLLBACKTRAN;
			MESSAGE BAS E2000 WITH 'VERİ EKLERKEN HATA OLUŞTU';
		ELSE
			MESSAGE BAS I2000 WITH 'VERİ EKLENDİ';
		ENDIF;
	ENDIF;

	IF EDUT2036TABLE1_UPDATED == 1 && EDUT2036TABLE1_DELETED == 0 THEN
		UPDATE EDUT2036TABLE1
			WHERE PERSID = EDUT2036TABLE1_PERSID;

		IF SYS_STATUS == 1 THEN
			ROLLBACKTRAN;
			MESSAGE BAS E2000 WITH 'VERİ GÜNCELLENEMEDİ';
		ELSE
			MESSAGE BAS I2000 WITH 'VERİ GÜNCELLENDİ';
		ENDIF;
	ENDIF;
	
	IF EDUT2036TABLE1_DELETED == 1 && EDUT2036TABLE1_READ == 1 THEN
		DELETE
			FROM EDUT2036TABLE1
			WHERE PERSID = EDUT2036TABLE1_ID;

		IF SYS_STATUS == 1 THEN
			ROLLBACKTRAN;
			MESSAGE BAS E2000 WITH 'VERİ SİLİNEMEDİ';
		ELSE
			MESSAGE BAS I2000 WITH 'VERİ SİLİNDİ';
		ENDIF;
	ENDIF;
ENDLOOP;
COMMITTRAN;
MESSAGE BAS I2000 WITH 'İŞLEM BAŞARILI';
